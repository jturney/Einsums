# Generate rst files with CMake variables and toolchains
set(EINSUMS_CMAKE_OPTIONS_RST "")
foreach(_cat ${EINSUMS_OPTION_CATEGORIES})
  string(REPLACE " " "_" _cat_link "${_cat}")
  set(EINSUMS_CMAKE_OPTIONS_RST
      "${EINSUMS_CMAKE_OPTIONS_RST}* :ref:`${_cat} options <${_cat_link}>`\n"
  )
endforeach()
set(EINSUMS_CMAKE_OPTIONS_RST "${EINSUMS_CMAKE_OPTIONS_RST}\n")

get_cmake_property(_variableNames CACHE_VARIABLES)
foreach(_cat ${EINSUMS_OPTION_CATEGORIES})
  string(REPLACE " " "_" _cat_anchor "${_cat}")
  set(_cat_title "${_cat} options")
  string(REGEX REPLACE "." "-" _cat_title_underline "${_cat_title}")
  set(EINSUMS_CMAKE_OPTIONS_RST
      "${EINSUMS_CMAKE_OPTIONS_RST}.. _${_cat_anchor}:\n\n${_cat_title}\n${_cat_title_underline}\n\n"
  )
  foreach(_variableName ${_variableNames})
    if(${_variableName}Category)
      if(_cat STREQUAL ${_variableName}Category)
        string(REPLACE " " "_" _variableName_anchor "${_variableName}")

        get_property(
          _doc
          CACHE "${_variableName}"
          PROPERTY HELPSTRING
        )
        get_property(
          _type
          CACHE "${_variableName}"
          PROPERTY TYPE
        )
        if(NOT _type STREQUAL "UNINITIALIZED")
          set(_variableName "${_variableName}:${_type}")
        endif()
        set(EINSUMS_CMAKE_OPTIONS_RST "${EINSUMS_CMAKE_OPTIONS_RST}* :option:`${_variableName}`\n")
      endif()
    endif()
  endforeach()
  set(EINSUMS_CMAKE_OPTIONS_RST "${EINSUMS_CMAKE_OPTIONS_RST}\n")

  foreach(_variableName ${_variableNames})
    if(${_variableName}Category)
      if(_cat STREQUAL ${_variableName}Category)
        string(REPLACE " " "_" _variableName_anchor "${_variableName}")

        get_property(
          _doc
          CACHE "${_variableName}"
          PROPERTY HELPSTRING
        )
        get_property(
          _type
          CACHE "${_variableName}"
          PROPERTY TYPE
        )
        if(NOT _type STREQUAL "UNINITIALIZED")
          set(_variableName "${_variableName}:${_type}")
        endif()
        set(EINSUMS_CMAKE_OPTIONS_RST
            "${EINSUMS_CMAKE_OPTIONS_RST}.. option:: ${_variableName}\n\n   ${_doc}\n\n"
        )
      endif()
    endif()
  endforeach()
endforeach()

# Copy the Sphinx source files to build directory.
#
# CONFIGURE_DEPENDS re-globs on every build and re-runs configure when the set
# of source files changes (a file is added or deleted) — without it, a new page
# needs a manual reconfigure and a deleted page is never noticed.
file(
  GLOB_RECURSE sphinx_source_files
  LIST_DIRECTORIES false CONFIGURE_DEPENDS
  RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}/"
  "sphinx/*"
)

# exclude api.rst...
list(FILTER sphinx_source_files EXCLUDE REGEX "sphinx/api/full_api.rst")

foreach(sphinx_source_file ${sphinx_source_files})
  add_custom_command(
    OUTPUT "${CMAKE_CURRENT_BINARY_DIR}/${sphinx_source_file}"
    DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/${sphinx_source_file}"
    COMMAND
      ${CMAKE_COMMAND} -E copy_if_different "${CMAKE_CURRENT_SOURCE_DIR}/${sphinx_source_file}"
      "${CMAKE_CURRENT_BINARY_DIR}/${sphinx_source_file}"
  )
endforeach()

foreach(sphinx_source_file ${sphinx_source_files})
  set(sphinx_source_files_build ${sphinx_source_files_build}
                                "${CMAKE_CURRENT_BINARY_DIR}/${sphinx_source_file}"
  )
  list(APPEND _copied_dest_files "${CMAKE_CURRENT_BINARY_DIR}/${sphinx_source_file}")
endforeach()

# Harden against stale copies. The copy step above is additive: when a
# hand-written source .rst is DELETED, its previously-copied file lingers in the
# build tree and Sphinx still builds it (this is how a removed page kept
# erroring on incremental builds). ``_copied_dest_files`` is exactly the set we
# copy this configure; compare it to the manifest written last configure and
# remove any orphan. Generated pages (api/, reference/python/, conf.py,
# CMakeVariables, cppdocs) are never in this set, so they are never touched.
# CONFIGURE_DEPENDS above guarantees a deletion re-triggers configure, so this
# prune actually runs when the source set shrinks.
set(_copied_manifest "${CMAKE_CURRENT_BINARY_DIR}/.sphinx_copied.manifest")
if(EXISTS "${_copied_manifest}")
  file(STRINGS "${_copied_manifest}" _prev_copied_dest_files)
  foreach(_old IN LISTS _prev_copied_dest_files)
    if(NOT "${_old}" IN_LIST _copied_dest_files)
      file(REMOVE "${_old}")
    endif()
  endforeach()
endif()
string(REPLACE ";" "\n" _manifest_text "${_copied_dest_files}")
file(WRITE "${_copied_manifest}" "${_manifest_text}\n")

# Generated Python API reference. The PyEinsumsDocs target runs
# ``apiary --emit-docs-json`` per module and renders the result to
# reStructuredText (grouped by submodule) via render_docs_rst.py. Copy the
# rendered pages into the Sphinx reference tree so python_module.rst's
# toctree can include reference/python/index. Additive for now: the
# hand-written einsums.core.* pages remain until their curated content is
# migrated into the C++ doc comments.
if(TARGET PyEinsumsDocs)
  set(_gen_py_docs_src    "${CMAKE_BINARY_DIR}/generated/pybind/docs")
  set(_gen_py_docs_dst    "${CMAKE_CURRENT_BINARY_DIR}/sphinx/reference/python")
  set(_gen_py_docs_stamp  "${CMAKE_CURRENT_BINARY_DIR}/.generated_python_docs.stamp")
  # Depend on the render's *output stamp file* (not the PyEinsumsDocs target):
  # a target dependency only enforces ordering, so the copy would not re-run
  # when the rendered pages change. The file's mtime is what ninja tracks.
  set(_render_stamp       "${CMAKE_BINARY_DIR}/generated/pybind/.docs.stamp")
  add_custom_command(
    OUTPUT "${_gen_py_docs_stamp}"
    DEPENDS PyEinsumsDocs "${_render_stamp}"
    COMMAND ${CMAKE_COMMAND} -E make_directory "${_gen_py_docs_dst}"
    COMMAND ${CMAKE_COMMAND} -E copy_directory "${_gen_py_docs_src}" "${_gen_py_docs_dst}"
    COMMAND ${CMAKE_COMMAND} -E touch "${_gen_py_docs_stamp}"
    COMMENT "docs: copying generated Python API reference into the Sphinx tree"
  )
  list(APPEND sphinx_source_files_build "${_gen_py_docs_stamp}")
endif()

foreach(output_format ${EINSUMS_WITH_DOCUMENTATION_OUTPUT_FORMATS})
  string(TOUPPER ${output_format} output_format_upper_case)
  set(SPHINX_DOCS_OUTPUT_DIR "${PROJECT_BINARY_DIR}/share/Einsums/docs/${output_format}")
  set(SPHINX_DOCS_OUTPUT_FILE
      "${SPHINX_DOCS_OUTPUT_DIR}/${SPHINX_DOCS_${output_format_upper_case}_OUTPUT_FILE}"
  )

  # NOTE: The PDF does not build without errors but still produces a reasonable PDF. If there are
  # real errors this will ignore them. Fixing the errors is at the moment not worth the effort.
  if(${output_format} STREQUAL "latexpdf")
    set(SPHINX_DOCS_BUILD_COMMAND
        ${Sphinx_EXECUTABLE} -M ${output_format} "${CMAKE_CURRENT_BINARY_DIR}/sphinx"
        "${SPHINX_DOCS_OUTPUT_DIR}" || (exit 0)
    )
  else()
    # -n  nitpicky (every dangling cross-reference is a warning)
    # -W  treat warnings as errors  --keep-going  report them all, not just the first
    set(SPHINX_DOCS_BUILD_COMMAND
        ${Sphinx_EXECUTABLE} -b ${output_format} -n -W --keep-going -d
        "${CMAKE_CURRENT_BINARY_DIR}/doctree" "${CMAKE_CURRENT_BINARY_DIR}/sphinx"
        "${SPHINX_DOCS_OUTPUT_DIR}"
    )
  endif()

  # The reference is generated statically from headers by apiary, so the
  # docs depend on the codegen stamps in ``sphinx_source_files_build`` (which pull
  # in the apiary tool and PyEinsumsDocs) — NOT on the compiled Einsums
  # library or the PyEinsums (_core) extension. Depending on those would force a
  # full library + extension build for no reason.
  add_custom_command(
    OUTPUT "${SPHINX_DOCS_OUTPUT_FILE}"
    DEPENDS "${sphinx_source_files_build}"
    COMMAND ${SPHINX_DOCS_BUILD_COMMAND}
  )

  add_custom_target(docs-${output_format}-file ALL DEPENDS ${SPHINX_DOCS_OUTPUT_FILE})
  add_dependencies(docs-${output_format} docs-${output_format}-file)

  add_dependencies(docs docs-${output_format})
endforeach()
